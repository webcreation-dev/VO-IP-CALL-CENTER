# ==============================================
# QUEUES MODULE
# ==============================================
# Base URL: http://localhost:3001/api/v1
# REMPLACER <TOKEN> par votre JWT token

# ============== LISTE DES QUEUES ==============
GET http://localhost:3001/api/v1/queues
Authorization: Bearer <TOKEN>


# ============== QUEUES ENRICHIES (avec stats AMI) ==============
GET http://localhost:3001/api/v1/queues/enriched
Authorization: Bearer <TOKEN>


# ============== STATS GLOBALES TOUTES QUEUES ==============
GET http://localhost:3001/api/v1/queues/stats/global
Authorization: Bearer <TOKEN>


# ============== QUEUES D'UN TENANT ==============
GET http://localhost:3001/api/v1/tenants/1/queues
Authorization: Bearer <TOKEN>


# ============== CRÉER QUEUE "support" (TENANT 1) ==============
POST http://localhost:3001/api/v1/queues
Authorization: Bearer <TOKEN>
Content-Type: application/json

{
  "tenantId": 1,
  "name": "support",
  "description": "Support client",
  "strategy": "ringall",
  "timeout": 15,
  "retry": 5,
  "wrapuptime": 10,
  "maxlen": 20
}


# ============== CRÉER QUEUE "sales" (TENANT 1) ==============

Authorization: Bearer <TOKEN>
Content-Type: application/json

{
  "tenantId": 1,
  "name": "sales",
  "description": "Équipe commerciale",
  "strategy": "leastrecent",
  "timeout": 20,
  "retry": 3,

GET http://localhost:3001/api/v1/queues/support

# ============== GET STATS TEMPS RÉEL (AMI) ==============
GET http://localhost:3001/api/v1/queues/support/stats

# ============== GET DÉTAILS COMPLETS (avec AMI) ==============
GET http://localhost:3001/api/v1/queues/support/details
Authorization: Bearer <TOKEN>


# ============== GET APPELS EN ATTENTE ==============
GET http://localhost:3001/api/v1/queues/support/calls

# ============== MODIFIER QUEUE ==============
PATCH http://localhost:3001/api/v1/queues/support
Authorization: Bearer <TOKEN>
Content-Type: application/json

{
  "description": "Support client - Mis à jour",
  "timeout": 20,
  "maxlen": 25
}


# ============== RECHARGER CONFIG QUEUE (AMI) ==============
POST http://localhost:3001/api/v1/queues/support/reload
Authorization: Bearer <TOKEN>


# ============== SUPPRIMER QUEUE ==============
DELETE http://localhost:3001/api/v1/queues/support
Authorization: Bearer <TOKEN>
